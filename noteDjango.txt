# install
1. sudo apt update
2. sudo apt install python3
3. sudo apt install python3-pip
4. pip install django
5. sudo apt install python3-django
6. django-admin startproject backend # to create and start a project  
7. python3 backend/manage.py migrate
8. python3 backend/manage.py startapp backend/src
9. sudo apt install python3.10-venv
10. python3 -m venv backend/

# pips
1. pip install psycopg2-binary

# run server 
1. python3 manage.py runserver
